\subsection{开发部署}
\begin{code}[title=/etc/nginx/sites-enabled/default]
	\begin{verbatim}
	sudo apt install -y nginx
server {
    listen 80 default_server;
    listen [::]:80 default_server;

    root /var/www/dist;

    index index.html index.htm index.nginx-debian.html;

    server_name your-domain.com;

    location / {
        try_files $uri $uri/ /index.html;
    }

    # 将 /api/auth/ 的请求转发到 keycloak.com/auth/
    location /api/auth/ {
            rewrite ^/api/auth/(.*)$ /auth/$1 break;

            proxy_pass https://keycloak.com;

            proxy_set_header Host keycloak.com;
            proxy_set_header X-Real-IP $remote_addr;
        }

    # 将 /api/v2/ 的请求转发到 data.com/
     location /api/v2/ {
            proxy_pass https://data.com;

            proxy_set_header Host data.com;
            proxy_set_header X-Real-IP $remote_addr;
        }

}
\end{verbatim}
\end{code}

\begin{code}[title=Build APK]
	\begin{verbatim}
	keytool -genkey -v -keystore ~/upload-keystore.jks -keyalg RSA -keysize 2048 -validity 10000 -alias upload
    java -jar uber-apk-signer-1.3.0.jar --apks game.apk
    java -jar uber-apk-signer-1.3.0.jar -a game.apk --ks upload-keystore.jks --ksAlias upload
    sudo waydroid app install app-universal-release-unsigned.apk
\end{verbatim}
\end{code}
