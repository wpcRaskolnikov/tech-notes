\section{远程桌面}
\begin{code}[title=relay]
	\begin{verbatim}
services:
  hbbs: # RustDesk ID/Rendezvous 服务器
    container_name: hbbs
    ports:
      - 21115:21115 # 用于 NAT 类型测试的 TCP
      - 21116:21116 # TCP打孔
      - 21116:21116/udp # UDP心跳/ID服务器
    image: rustdesk/rustdesk-server:latest
    command: hbbs
    volumes:
      - /data/rustdesk/hbbs:/root
    environment:
      - 'RELAY=192.168.143.6:21117' # 运行这些容器的服务器的【IP:port】或域名
      - 'ENCRYPTED_ONLY=1' # 开启加密
      - 'KEY=Key' # 自定义KEY，去掉这一行可以自动生成
    depends_on:
      - hbbr
    restart: unless-stopped

  hbbr: # RustDesk 中继服务器
    container_name: hbbr
    ports:
      - 21117:21117 # TCP中继
    image: rustdesk/rustdesk-server:latest
    command: hbbr
    volumes:
      - /data/rustdesk/hbbr:/root
    restart: unless-stopped
    \end{verbatim}
\end{code}

\begin{code}[title=server]

# 走中继
sudo mkdir -p ~/.config/rustdesk
sudo tee ~/.config/rustdesk/RustDesk2.toml > /dev/null <<'EOF'
[options]
custom-rendezvous-server = "192.168.143.6"
relay-server = "192.168.143.6"
key = "Key"
direct-server = "Y"
direct-access-port = "21118"
EOF

rustdesk --get-id
rustdesk --password <your_password>
    \end{verbatim}
\end{code}
